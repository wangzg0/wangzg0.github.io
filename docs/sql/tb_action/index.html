数据库操作

创建表#

CREATE TABLE `table_user`(
`id` INT(1) NOT NULL UNSIGNED AUTO_INCREMENT,
`name` VARCHAR(20) NOT NULL,
`email` VARCHAR(50) NOT NULL,
`dssc` TEXT,
`create_time` TIMESTAMP,
PRIMARY_KEY(`id`)
)ENGIN=InnoDB DEFAULT CHARSET=utf8;
  • NOT NULL： 不允许空值
  • AUTO_INCREMENT 自增
  • ENGIN 设置存储引擎
  • CHARSET 设置数据编码
  • PRIMARY_KEY 设置主键
  • UNSIGNED 无符号

删除表#

DROP TABLE `table_user`;

数据插入#

INSERT INTO `table_user`(field1, field2, field3, ... fieldN) VALUES (value1, value2, value3, ...valueN);

数据查询#

# 批量查询
SELECT * FROM `table_user`;
# 查询某个字段列表
SELECT name FROM `table_user`;
# 过滤查询
SELECT `name` FROM `table_user` WHERE id = 1;
